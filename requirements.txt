# ═══════════════════════════════════════
# DÉPLOIEMENT RAILWAY — étape par étape
# ═══════════════════════════════════════

# 1. Installer Railway CLI
npm install -g @railway/cli

# 2. Init projet
railway login
railway init   # choisir "Empty project"

# 3. Lier le dossier
cd analyseur-dce
railway link

# 4. Ajouter la clé API (JAMAIS dans le code)
railway variables set ANTHROPIC_API_KEY=sk-ant-XXXXXXXX

# 5. Déployer
railway up

# → Railway te donne une URL publique type :
#   https://analyseur-dce-production.up.railway.app
